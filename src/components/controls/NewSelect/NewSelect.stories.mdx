# NewSelect

### Uncontrolled

<Story
  name="NewSelect"
  args={{
    isMobile: false,
    closeOnClear: false,
    multiple: false,
    disabled: false,
    circularNavigation: true,
    closeOnSelect: true,
    labelView: 'inner',
    options: [
      {
        key: 'true',
        content: 'True value',
        value: true,
      },
      {
        key: 'false',
        content: 'False value',
        value: false,
      },
      {
        key: 'zero',
        content: <i>Zero value</i>,
        value: 0,
      },
      {
        key: 'empty string value',
        content: 'Empty string',
        value: '',
      },
      {
        key: 'tough content',
        content: <strong>You can use bold</strong>,
        value: 'bold',
      },
      {
        key: '1',
        content: '1',
        value: '1',
      },
      {
        key: '2',
        content: '2',
        value: '2',
        disabled: true,
      },
      {
        key: '3',
        content: '3',
        value: '3',
      },
      {
        key: '4',
        content: '4',
        value: '4',
      },
    ],
  }}
  argTypes={{
    labelView: {
      options: ['inner', 'outer'],
      control: { type: 'radio' },
    },
  }}
>
  {({ closeOnClear, ...args }) => {
    const [value, setValue] = useState();
    const [open, setOpen] = useState(false);
    const clearSelectProps = useSelectClear({
      closeOnClear,
    });
    const clearableItemsProps = useClearableItems({
      Field: clearSelectProps.Field,
    });
    return (
      <div style={{ width: 500 }}>
        <p>
          Выбрано значение:{' '}
          <b>{value === undefined ? '(undefined)' : JSON.stringify(value)}</b>
        </p>
        <NewSelect
          {...args}
          onChange={(e) => {
            if (!args.multiple) {
              setValue(e.selected?.value);
            } else {
              setValue(e.selectedMultiple?.map((e) => e.value) || []);
            }
          }}
          open={open}
          onOpen={(payload) => {
            setOpen(payload.open);
          }}
          allowUnselect
          placeholder="Выберите"
          {...clearSelectProps}
          {...clearableItemsProps}
        />
        <Button css={{ marginTop: scale(1) }} onClick={() => setOpen(!open)}>
          {!open ? 'Открыть' : 'Закрыть'} вручную
        </Button>
      </div>
    );
  }}
</Story>
